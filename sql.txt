新建数据库：
CREATE DATABASE zulin

新建表：
设备表
create table device(
id int(5) PRIMARY key auto_increment,
name varchar(20)not null,
price decimal(6,2),
count integer(6),
last integer(6),
status tinyint(1),
unit varchar(10),
loss_price decimal(6,2),
PRIMARY KEY  (`id`)
)
客户表
create table site(
id int(5) PRIMARY key auto_increment,
name varchar(20)not null,
person_name varchar(10),
phone varchar(15),
status tinyint(1),
paid_money decimal(8,2),
PRIMARY KEY  (`id`)
)
订单流水表
create table flow(
id int(5) PRIMARY key auto_increment,
begin_date varchar(13),
create_date varchar(13),
number int(6),
type tinyint not null,
money decimal(8,2),
discount decimal(8,2),
status tinyint(1),
settle tinyint(0),
settle_last int(6),
site_id int(5),
device_id int(5),
detail varchar(500)
foreign key(site_id) references site(id),
foreign key(device_id) references device(id),
PRIMARY KEY  (`id`)
)
订单汇总表
create table summary(
id int(5) PRIMARY key auto_increment,
site_id int(5),
device_id int(5),
foreign key(site_id) references site(id),
foreign key(device_id) references device(id),
total int(6),
revert int(6),
PRIMARY KEY  (`id`)
)
用户表
create table user(
id int(5) PRIMARY key auto_increment,
name varchar(20),
password varchar(32),
type tinyint(1),
site_id int(5),
PRIMARY KEY  (`id`),
status tinyint(1)
)
